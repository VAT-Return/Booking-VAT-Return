<script>

// ✅ STORAGE
let payments = JSON.parse(localStorage.getItem("payments") || "{}");

// ✅ OR GENERATOR
function generateOR(year){
  let count = Object.keys(payments).filter(k=>k.startsWith(year)).length + 1;
  return "OR-"+year+"-"+String(count).padStart(4,'0');
}

// ✅ FORMAT
function peso(n){
  return "₱ "+Number(n).toLocaleString("en-PH",{minimumFractionDigits:2});
}

let currentYear="2026";

// ✅ DATA
const data2026=[
["January",4284,514,"Paid","Q1"],
["February",9240,1108,"Paid","Q1"],
["March",2396,287,"Paid","Q1"],
["April",756,90,"Paid","Q2"],
["May",10701,1284,"Unpaid","Q2"]
];

const data2025={
January:2511, February:14094, March:3987, April:4770.9,
May:5859, June:753.3, September:837,
October:4185, November:1674, December:3180.6
};

const months=["January","February","March","April","May","June","July","August","September","October","November","December"];

// LOGIN
function login(){
if(user.value==="sales@bellagiohills.com.ph" &&
pass.value==="GMBellagiohills#2025"){
loginPage.style.display="none";
sidebar.style.display="block";
main.style.display="block";
load("2026");
}
}

// LOAD
function load(year){
currentYear=year;

let tb=vatTable.querySelector("tbody");
tb.innerHTML="";

let total=0, paidCount=0, unpaidCount=0;

// ✅ 2026
if(year==="2026"){
data2026.forEach(r=>{
let key = year+"-"+r[0];
let isPaid = payments[key]?.paid || r[3]=="Paid";

let status = isPaid ? "Paid" : "Unpaid";

if(isPaid) paidCount++; else unpaidCount++;

total+=r[2];

tb.innerHTML+=`
<tr data-q="${r[4]}">
<td>${r[0]}</td>
<td>${peso(r[1])}</td>
<td>${peso(r[2])}</td>
<td class="${status=='Paid'?'paid':'unpaid'}">${status}</td>
<td>${r[4]}</td>
<td>${isPaid
? `<button class="view-btn" onclick="viewPayment('${year}','${r[0]}',${r[1]},${r[2]})">View</button>`
: `<button class="view-btn" onclick="payNow('${year}','${r[0]}',${r[1]},${r[2]})">Pay</button>`
}
</td>
</tr>`;
});

["June","July","August","September","October","November","December"].forEach(m=>{
tb.innerHTML+=`<tr><td>${m}</td><td colspan="5">Unavailable</td></tr>`;
});
}

// ✅ 2025
if(year==="2025"){
months.forEach(m=>{
let c=data2025[m]||0;
let v=c*0.12;

if(c>0){ total+=v; unpaidCount++; }

tb.innerHTML+=`
<tr><td>${m}</td>
<td>${peso(c)}</td>
<td>${peso(v)}</td>
<td class="unpaid">Unpaid</td>
<td>-</td>
<td>Unsubmitted</td>
</tr>`;
});
}

// DASHBOARD
tot.innerText=peso(total);
paid.innerText=paidCount;
unpaid.innerText=unpaidCount;
}

// ✅ PAY
function payNow(year,month,comm,vat){
let key=year+"-"+month;

let orNum=generateOR(year);

payments[key]={paid:true, or:orNum};

localStorage.setItem("payments",JSON.stringify(payments));

alert("Payment successful\n"+orNum);

load(year);
}

// ✅ VIEW
function viewPayment(year,month,comm,vat){
let key=year+"-"+month;

mMonth.innerText="Month: "+month;
mCommission.innerText="Commission: "+peso(comm);
mVAT.innerText="VAT: "+peso(vat);

statusText.innerText="✔ PAID";
orText.innerText="OR Number: "+payments[key].or;

modal.style.display="block";
}

function closeModal(){ modal.style.display="none"; }

// FILTER
function filterQuarter(q){
document.querySelectorAll("#vatTable tbody tr").forEach(r=>{
if(!r.dataset.q) return;
r.style.display=(q=="all"||r.dataset.q==q)?"":"none";
});
}
